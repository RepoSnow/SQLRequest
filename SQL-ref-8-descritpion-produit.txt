insert into ref_description_maj 

	select a.CN,
	CONCAT( 
	CASE WHEN f.DESCRIPTION IS NOT NULL THEN f.DESCRIPTION+', ' end,
	CASE WHEN e.DESCRIPTION IS NOT NULL THEN e.DESCRIPTION+', 'end,
	CASE WHEN d.DESCRIPTION IS NOT NULL THEN d.DESCRIPTION+', 'end,
	CASE WHEN c.DESCRIPTION IS NOT NULL THEN c.DESCRIPTION+', ' end,
	CASE WHEN b.DESCRIPTION IS NOT NULL THEN b.DESCRIPTION+', 'end,
    a.DESCRIPTION
	) DESCRIPTION
	from ref_detail_description a
	left JOIN ref_detail_description b on left(a.CN,7) = b.CN and b.LENGHT =7
		left JOIN ref_detail_description c on left(a.CN,6) = c.CN and c.LENGHT  =6
			left JOIN ref_detail_description d on left(a.CN,5) = d.CN and d.LENGHT  =5
				left JOIN ref_detail_description e on left(a.CN,4) = left(e.CNKEY,4) and e.CN is null  
					left JOIN ref_detail_description f on left(a.CN,4) = left(f.CN,4) and f.LENGHT  =4 
	where a.LENGHT = 8
	order by a.CN;
